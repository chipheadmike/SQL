## Order Status Listing
```sql
SELECT PHCMDT AS SBD,
sum(CASE WHEN PHPSTF='10' THEN 1 ELSE 0 END) AS Unselected,
sum(CASE WHEN PHPSTF='15' THEN 1 ELSE 0 END) AS Preview,
sum(CASE WHEN PHPSTF='20' AND WDWVST='15' THEN 1 ELSE 0 END) AS PrintedNOTReleased,
sum(CASE WHEN PHPSTF='20' AND WDWVST='20' THEN 1 ELSE 0 END) AS PrintedReleased,
sum(CASE WHEN PHPSTF='35' THEN 1 ELSE 0 END) AS InPacking,
sum(CASE WHEN PHPSTF='40' THEN 1 ELSE 0 END) AS PackedComplete
FROM WM370BASD.PHPICK00 JOIN WM370BASD.WDWAVE00 ON PHPRWV=WDPRWV
WHERE PHCO='2760' GROUP BY PHCMDT ORDER BY 1 DESC;

SUM(CASE WHEN PHCMDT='20210211' then 1 else 0 end) as Thursday,
SUM(CASE WHEN PHCMDT='20210212' then 1 else 0 end) as Friday
```
